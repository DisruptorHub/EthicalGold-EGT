                        const delay = baseDelay * (2 ** retryCount);

                        await new Promise(res => setTimeout(res, delay));

                        makeApiCall(retryCount + 1);

                    } else {

                        traceResponseDiv.textContent = `An error occurred. Please check the console for details. Make sure you have added a valid API key.`;

                    }

                }

            };

            

            makeApiCall();

        };
